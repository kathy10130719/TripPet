// 測試開啟相機的功能 async function startCamera() { try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } }); const videoElement = document.getElementById('cameraFeed'); videoElement.srcObject = stream; // 這時候背景就會變成相機畫面，Kiki 依然浮在上面 } catch (err) { alert("無法開啟相機，請檢查權限設定喔！"); } }
